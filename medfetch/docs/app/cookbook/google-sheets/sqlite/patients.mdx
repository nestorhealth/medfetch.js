```sql copy
CREATE TABLE IF NOT EXISTS abc_patients AS
SELECT DISTINCT
  p.id AS id,
  p.json AS json
FROM medfetch('Patient') p
INNER JOIN acute_bronchitis_conditions c ON p.json->>'id' = c.patient_id;

SELECT * FROM abc_patients;
```
